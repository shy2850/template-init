使用模板加载的数据列表携带分页

点击分页页码可以发送ajax请求



(function(T){
    T.init({
        tmpl: "#tmpl-01",
        target: "#show-01",
        sourceUrl: "data/list.json",
        sourceMethod: "get",
        dataType: "json", // jsonp 也可以
        page:{size:'page.size',to:'page.pn'},  //提交分页相关参数名称配置
        pagination: ".pager"    // 分页结果填充标签
    });
})(TemplateInit);
